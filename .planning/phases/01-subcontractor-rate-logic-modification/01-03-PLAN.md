    ```python
    def _resolve_row_price(row, variant, missing_cus):
        """Return the per-row Units Total Price to write to Excel.

        Per D-14/D-15: for primary / helper / vac_crew, return the
        existing SmartSheet ``Units Total Price`` value unchanged.
        Per D-08/D-16: for aep_billable / reduced_sub (and helper
        shadow forms), look up the CU code + work type in
        ``_SUBCONTRACTOR_RATES`` and return ``rate × qty``. If the
        CU is missing, fall through to SmartSheet pricing
        (NEVER zero-out, NEVER raise) and record the missing code in
        ``missing_cus`` (Counter[str]).

        Canonical column names per ``_validate_single_sheet`` at
        L2523-2547 (verified by Blocker 2 read-first):
          - ``row['CU']``              (canonical for CU code)
          - ``row['Work Type']``       (canonical for work type)
          - ``row['Quantity']``        (canonical for qty)
          - ``row['Units Total Price']`` (canonical for price fallback)
        Reading any other key would be a silent regression — only
        the canonical keys exist at this point in the pipeline.
        """
        if variant not in ('aep_billable', 'reduced_sub',
                           'aep_billable_helper', 'reduced_sub_helper'):
            return parse_price(row.get('Units Total Price'))
        # CU lookup uses canonical 'CU' ONLY (synonyms 'CU' and
        # 'Billable Unit Code' both collapse to row['CU'] upstream).
        cu_raw = row.get('CU') or ''
        cu = str(cu_raw).strip().upper()
        rate_row = _SUBCONTRACTOR_RATES.get(cu)
        if rate_row is None:
            if cu:
                missing_cus[cu] += 1
            return parse_price(row.get('Units Total Price'))
        # Work-type-keyed column selection (D-05) — canonical key
        # 'Work Type' only.
        work_type_raw = (row.get('Work Type') or '').strip().lower()
        if 'install' in work_type_raw:
            wt = 'install'
        elif 'remov' in work_type_raw:
            wt = 'remove'
        elif 'transfer' in work_type_raw:
            wt = 'transfer'
        else:
            # Unknown work type — fall through to SmartSheet (safe)
            return parse_price(row.get('Units Total Price'))
        if variant in ('aep_billable', 'aep_billable_helper'):
            rate = rate_row.get(f'new_{wt}_price', 0.0)
        else:  # reduced_sub / reduced_sub_helper
            rate = rate_row.get(f'reduced_{wt}_price', 0.0)
        # Quantity lookup uses canonical 'Quantity' ONLY (synonyms
        # 'Qty', '# Units' all collapse to row['Quantity'] upstream).
        # Do NOT read 'Units Completed' — that's the boolean
        # checkbox 'Units Completed?'.
        qty_raw = row.get('Quantity') or 0
        try:
            qty = float(qty_raw)
        except (TypeError, ValueError):
            qty = 0.0
        if rate <= 0 or qty <= 0:
            # Degenerate row — keep SmartSheet pricing as the
            # safety floor; do NOT silently zero-out.
            return parse_price(row.get('Units Total Price'))
        return rate * qty
    ```
