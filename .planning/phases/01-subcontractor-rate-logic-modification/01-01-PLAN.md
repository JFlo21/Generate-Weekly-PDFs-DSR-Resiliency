    ```python
    def load_subcontractor_rates(filepath: str) -> dict[str, dict]:
        """Load the subcontractor rates CSV into a CU-keyed dict.

        Per D-04..D-07: tolerates UTF-8 BOM, strips ``$`` and thousands-
        comma from price cells via ``parse_price``, skips rows whose six
        priced columns are all zero (placeholder/inactive CUs), reads
        literal values (no ``reduced = old × 0.87`` shortcut math), does
        NOT load Old-Rates columns 12-14 or Hours columns 6-8.

        Returns ``{}`` on failure (fail-safe contract); never raises.
        """
        rates: dict[str, dict] = {}
        REQUIRED = {
            'CU',
            'Install Price (Subcontractor Rates)',
            'Removal Price (Subcontractor Rates)',
            'Transfer Price (Subcontractor Rates)',
            'Install Price (New Rates)',
            'Removal Price (New Rates)',
            'Transfer Price (New Rates)',
        }
        try:
            with open(filepath, mode='r', encoding='utf-8-sig', newline='') as f:
                reader = csv.DictReader(f)
                fieldnames = set(reader.fieldnames or [])
                missing = REQUIRED - fieldnames
                if missing:
                    logging.error(
                        f"Subcontractor rates CSV {filepath} missing "
                        f"required headers: {missing}"
                    )
                    return rates
                for row in reader:
                    cu = str(row.get('CU', '') or '').strip().upper()
                    if not cu:
                        continue
                    reduced_install = parse_price(
                        row.get('Install Price (Subcontractor Rates)', 0))
                    reduced_remove = parse_price(
                        row.get('Removal Price (Subcontractor Rates)', 0))
                    reduced_transfer = parse_price(
                        row.get('Transfer Price (Subcontractor Rates)', 0))
                    new_install = parse_price(
                        row.get('Install Price (New Rates)', 0))
                    new_remove = parse_price(
                        row.get('Removal Price (New Rates)', 0))
                    new_transfer = parse_price(
                        row.get('Transfer Price (New Rates)', 0))
                    # D-04: skip rows whose all six priced columns are zero
                    if (reduced_install == 0 and reduced_remove == 0
                            and reduced_transfer == 0
                            and new_install == 0 and new_remove == 0
                            and new_transfer == 0):
                        continue
                    rates[cu] = {
                        'cu_code': cu,
                        'cu_wbs': str(row.get('CU WBS #', '') or '').strip(),
                        'compatible_unit_group': str(
                            row.get('Compatible Unit Group', '') or '').strip(),
                        'reduced_install_price': reduced_install,
                        'reduced_remove_price': reduced_remove,
                        'reduced_transfer_price': reduced_transfer,
                        'new_install_price': new_install,
                        'new_remove_price': new_remove,
                        'new_transfer_price': new_transfer,
                    }
            logging.info(
                f"Loaded {len(rates)} subcontractor CU rates from {filepath}"
            )
        except Exception as e:  # noqa: BLE001 — fail-safe contract
            logging.error(
                f"Failed to load subcontractor rates from {filepath}: {e}"
            )
        return rates


    def _compute_subcontractor_rates_fingerprint(rates_dict: dict[str, dict]) -> str:
        """Per D-20: deterministic 16-char SHA256 prefix over all six priced
        fields. Sorted keys + fixed precision guarantees byte-identical
        output for byte-identical input across runs and across machines.
        """
        h = hashlib.sha256()
        for code in sorted(rates_dict.keys()):
            r = rates_dict[code]
            h.update(
                f"{code}:"
                f"{r['reduced_install_price']:.2f},"
                f"{r['reduced_remove_price']:.2f},"
                f"{r['reduced_transfer_price']:.2f},"
                f"{r['new_install_price']:.2f},"
                f"{r['new_remove_price']:.2f},"
                f"{r['new_transfer_price']:.2f}\n".encode()
            )
        return h.hexdigest()[:16]
    ```
